### 代码中对应的判断堆栈

在你的 `ScalePlayerDamage` 钩子中，可以按如下伪代码实现上述逻辑，并打印详细堆栈：

```lua
hook.Add("ScalePlayerDamage", "EPDA_HandleDamage", function(ply, hitgroup, dmginfo)
    local ctx = ply.EPDA_Context
    local dmg = dmginfo:GetDamage()
    local hpBefore = ply:Health()
    local addBefore = ctx.additionalHealth

    print(string.format("[EPDA] >> OnDamage: %d", dmg))
    print(string.format("[EPDA]    [Pre] HP:%d Add:%d Logic:%d", hpBefore, addBefore, hpBefore+addBefore))

    local engineAfter = hpBefore - dmg
    if engineAfter > 0 then
        print("[EPDA]    [Check] Engine can absorb => no realization")
        -- 正常结算，无需干预
        -- 但需要根据结算后的逻辑血量判断状态是否改变（例如 Combat 内逻辑血量下降，但仍大于75则保持 Combat）
        local logicAfter = engineAfter + addBefore
        print(string.format("[EPDA]    [Predict] LogicAfter:%d", logicAfter))
        -- 状态转移判断（仅更新状态，不兑现额外血量）
        -- ...
    else
        print("[EPDA]    [Check] EngineHealthAfter <=0 => need realization")
        local logicRemaining = (hpBefore + addBefore) - dmg
        print(string.format("[EPDA]    [LogicCalc] (%d + %d) - %d = %d", hpBefore, addBefore, dmg, logicRemaining))

        local realized = 0
        if logicRemaining <= 0 then
            print("[EPDA]    [Realize] LogicRemaining <=0 => no realization, let engine kill")
        elseif logicRemaining <= ctx.writhingHealth then
            realized = addBefore
            print(string.format("[EPDA]    [Realize] Logic ∈ (0, %d] => realize all: %d", ctx.writhingHealth, realized))
        elseif logicRemaining <= ctx.writhingHealth + ctx.strugglingHealth then
            realized = ctx.strugglingHealth
            print(string.format("[EPDA]    [Realize] Logic ∈ (%d, %d] => realize strugglingHealth: %d",
                ctx.writhingHealth, ctx.writhingHealth+ctx.strugglingHealth, realized))
        end

        if realized > 0 then
            ply:SetHealth(hpBefore + realized)
            ctx.additionalHealth = addBefore - realized
            print(string.format("[EPDA]    [Realize] SetHealth(%d + %d) = %d | AddPool = %d",
                hpBefore, realized, ply:Health(), ctx.additionalHealth))
        end

        -- 伤害正常结算（引擎扣除 dmg）
        -- 在钩子结束后引擎会自动扣除，我们不做 ScaleDamage(0)
        -- 但需要预测结算后血量以进行状态转移
        local postHealth = ply:Health() - dmg
        local postAdd = ctx.additionalHealth
        local logicAfter = postHealth + postAdd
        print(string.format("[EPDA]    [Post] HP:%d Add:%d Logic:%d", postHealth, postAdd, logicAfter))

        -- 状态转移判断
        -- ...
    end
end)
```
